[HttpPost]
[Route("addrows")]
public async Task<IActionResult> AddRows(int startRow, int numberOfRows)
{
    var connectionString = _connectionString;
    var tableName = "employee_info";

    using (var connection = new MySqlConnection(connectionString))
    {
        await connection.OpenAsync();

        // Step 1: Shift existing rows down
        var shiftQuery = $"UPDATE `{tableName}` SET `RowNo` = `RowNo` + @NumberOfRows WHERE `RowNo` >= @StartRow";
        using (var shiftCommand = new MySqlCommand(shiftQuery, connection))
        {
            shiftCommand.Parameters.AddWithValue("@NumberOfRows", numberOfRows);
            shiftCommand.Parameters.AddWithValue("@StartRow", startRow);
            await shiftCommand.ExecuteNonQueryAsync();
        }

        // Step 2: Insert new rows
        for (int i = 0; i < numberOfRows; i++)
        {
            var insertQuery = $"INSERT INTO `{tableName}` (`RowNo`, `Column1`, `Column2`, ...) VALUES (@RowNo, @Value1, @Value2, ...)";
            using (var insertCommand = new MySqlCommand(insertQuery, connection))
            {
                insertCommand.Parameters.AddWithValue("@RowNo", startRow + i);
                insertCommand.Parameters.AddWithValue("@Value1", /* your value here */);
                insertCommand.Parameters.AddWithValue("@Value2", /* your value here */);
                // Add parameters for other columns as needed

                await insertCommand.ExecuteNonQueryAsync();
            }
        }

        return Ok(new { Status = true });
    }
}
